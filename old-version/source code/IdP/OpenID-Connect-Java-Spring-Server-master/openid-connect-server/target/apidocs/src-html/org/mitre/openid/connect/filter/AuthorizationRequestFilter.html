031import static org.mitre.openid.connect.request.ConnectRequestParameters.STATE;

071/**
072 * @author jricher
073 *
074 */
075@Component("authRequestFilter")
076public class AuthorizationRequestFilter extends GenericFilterBean {
077
078        /**
079         * Logger for this class
080         */
081        private static final Logger logger = LoggerFactory.getLogger(AuthorizationRequestFilter.class);
082
083        public final static String PROMPTED = "PROMPT_FILTER_PROMPTED";
084        public final static String PROMPT_REQUESTED = "PROMPT_FILTER_REQUESTED";
085
086        @Autowired
087        private OAuth2RequestFactory authRequestFactory;
088
089        @Autowired
090        private ClientDetailsEntityService clientService;
091
092        @Autowired
093        private RedirectResolver redirectResolver;
094
095        @Autowired(required = false)
096        private LoginHintExtracter loginHintExtracter = new RemoveLoginHintsWithHTTP();
097
098        private RequestMatcher requestMatcher = new AntPathRequestMatcher("/authorize");
099
100        /**
101         *
102         */
103        @Override
104        public void doFilter(ServletRequest req, ServletResponse res, FilterChain chain) throws IOException, ServletException {
105
106                HttpServletRequest request = (HttpServletRequest) req;
107                HttpServletResponse response = (HttpServletResponse) res;
108                HttpSession session = request.getSession();
109
110                // skip everything that's not an authorize URL
111                if (!requestMatcher.matches(request)) {
112                        
113                        
114                        chain.doFilter(req, res);
115                        
116                        return;
117                }
118
119                try {
120                        // we have to create our own auth request in order to get at all the parmeters appropriately
121                        AuthorizationRequest authRequest = null;
122
123                        ClientDetailsEntity client = null;
124
125                        authRequest = authRequestFactory.createAuthorizationRequest(createRequestMap(request.getParameterMap()));
126                        
127                        logger.info("this is test for clientID" + authRequest.getClientId());
128                        
129                        if (!Strings.isNullOrEmpty(authRequest.getClientId())) {
130                                client = clientService.loadClientByClientId(authRequest.getClientId());
131                        }
132                        
133                        
134                        // save the login hint to the session
135                        // but first check to see if the login hint makes any sense
136                        String loginHint = loginHintExtracter.extractHint((String) authRequest.getExtensions().get(LOGIN_HINT));
137                        if (!Strings.isNullOrEmpty(loginHint)) {
138                                
139                                
140                                session.setAttribute(LOGIN_HINT, loginHint);
141                        } else {
142                                
143                                
144                                session.removeAttribute(LOGIN_HINT);
145                        }
146
147                        if (authRequest.getExtensions().get(PROMPT) != null) {
148                                
149                                
150                                // we have a "prompt" parameter
151                                String prompt = (String)authRequest.getExtensions().get(PROMPT);
152                                List<String> prompts = Splitter.on(PROMPT_SEPARATOR).splitToList(Strings.nullToEmpty(prompt));
153
154                                if (prompts.contains(PROMPT_NONE)) {
155                                        
156                                        
157                                        // see if the user's logged in
158                                        Authentication auth = SecurityContextHolder.getContext().getAuthentication();
159
160                                        if (auth != null) {
161                                                // user's been logged in already (by session management)
162                                                // we're OK, continue without prompting
163                                                
164                                                
165                                                chain.doFilter(req, res);
166                                        } else {
167                                                
168                                                
169                                                logger.info("Client requested no prompt");
170                                                // user hasn't been logged in, we need to "return an error"
171                                                if (client != null && authRequest.getRedirectUri() != null) {
172                                                        
173                                                        
174
175                                                        // if we've got a redirect URI then we'll send it
176
177                                                        String url = redirectResolver.resolveRedirect(authRequest.getRedirectUri(), client);
178
179                                                        try {
180                                                                URIBuilder uriBuilder = new URIBuilder(url);
181
182                                                                uriBuilder.addParameter(ERROR, LOGIN_REQUIRED);
183                                                                if (!Strings.isNullOrEmpty(authRequest.getState())) {
184                                                                        
185                                                                        
186                                                                        uriBuilder.addParameter(STATE, authRequest.getState()); // copy the state parameter if one was given
187                                                                }
188
189                                                                response.sendRedirect(uriBuilder.toString());
190                                                                return;
191
192                                                        } catch (URISyntaxException e) {
193                                                                logger.error("Can't build redirect URI for prompt=none, sending error instead", e);
194                                                                response.sendError(HttpServletResponse.SC_FORBIDDEN, "Access Denied");
195                                                                return;
196                                                        }
197                                                }
198
199                                                
200                                                
201                                                response.sendError(HttpServletResponse.SC_FORBIDDEN, "Access Denied");
202                                                return;
203                                        }
204                                } else if (prompts.contains(PROMPT_LOGIN)) {
205
206                                        
207                                        // first see if the user's already been prompted in this session
208                                        if (session.getAttribute(PROMPTED) == null) {
209                                                // user hasn't been PROMPTED yet, we need to check
210                                                
211
212                                                session.setAttribute(PROMPT_REQUESTED, Boolean.TRUE);
213
214                                                // see if the user's logged in
215                                                Authentication auth = SecurityContextHolder.getContext().getAuthentication();
216                                                if (auth != null) {
217                                                        
218                                                        
219                                                        // user's been logged in already (by session management)
220                                                        // log them out and continue
221                                                        SecurityContextHolder.getContext().setAuthentication(null);
222                                                        chain.doFilter(req, res);
223                                                } else {
224                                                        // user hasn't been logged in yet, we can keep going since we'll get there
225                                                        
226                                                        
227                                                        chain.doFilter(req, res);
228                                                }
229                                        } else {
230                                                
231                                                
232                                                // user has been PROMPTED, we're fine
233
234                                                // but first, undo the prompt tag
235                                                session.removeAttribute(PROMPTED);
236                                                chain.doFilter(req, res);
237                                        }
238                                } else {
239                                        // prompt parameter is a value we don't care about, not our business
240                                        
241                                        
242                                        chain.doFilter(req, res);
243                                }
244
245                        } else if (authRequest.getExtensions().get(MAX_AGE) != null ||
246                                        (client != null && client.getDefaultMaxAge() != null)) {
247                                
248                                
249
250                                // default to the client's stored value, check the string parameter
251                                Integer max = (client != null ? client.getDefaultMaxAge() : null);
252                                String maxAge = (String) authRequest.getExtensions().get(MAX_AGE);
253                                if (maxAge != null) {
254                                        max = Integer.parseInt(maxAge);
255                                }
256
257                                if (max != null) {
258
259                                        Date authTime = (Date) session.getAttribute(AuthenticationTimeStamper.AUTH_TIMESTAMP);
260
261                                        Date now = new Date();
262                                        if (authTime != null) {
263                                                long seconds = (now.getTime() - authTime.getTime()) / 1000;
264                                                if (seconds > max) {
265                                                        // session is too old, log the user out and continue
266                                                        SecurityContextHolder.getContext().setAuthentication(null);
267                                                }
268                                        }
269                                }
270                                
271                                
272                                
273                                chain.doFilter(req, res);
274                        } else {
275                                // no prompt parameter, not our business
276                                
277                                chain.doFilter(req, res);
278                        }
279
280                } catch (InvalidClientException e) {
281                        // we couldn't find the client, move on and let the rest of the system catch the error
282                        chain.doFilter(req, res);
283                }
284        }
285
286        /**
287         * @param parameterMap
288         * @return
289         */
290        private Map<String, String> createRequestMap(Map<String, String[]> parameterMap) {
291                Map<String, String> requestMap = new HashMap<>();
292                for (String key : parameterMap.keySet()) {
293                        String[] val = parameterMap.get(key);
294                        if (val != null && val.length > 0) {
295                                requestMap.put(key, val[0]); // add the first value only (which is what Spring seems to do)
296                        }
297                }
298
299                return requestMap;
300        }
301
302        /**
303         * @return the requestMatcher
304         */
305        public RequestMatcher getRequestMatcher() {
306                return requestMatcher;
307        }
308
309        /**
310         * @param requestMatcher the requestMatcher to set
311         */
312        public void setRequestMatcher(RequestMatcher requestMatcher) {
313                this.requestMatcher = requestMatcher;
314        }
